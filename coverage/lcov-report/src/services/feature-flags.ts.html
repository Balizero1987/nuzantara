  ENABLE_AUTO_SCALING = 'enable_auto_scaling',
  ENABLE_SESSION_AFFINITY = 'enable_session_affinity',

  // Safety features
  ENABLE_GRADUAL_ROLLOUT = 'enable_gradual_rollout',
